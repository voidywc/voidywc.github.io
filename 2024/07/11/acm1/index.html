容斥原理小结1 —数论篇

容斥原理运用

容斥原理可以抽象出两类：

  1. 在有n个限制条件的情况下， 选出k（k <= n）个限制条件。根据选出限制条件的奇偶性来对答案进行加减，如：
    if (k ^ 1) ans += res; else ans -= res;
  2. 当需要求一个数以及它的因数的影响的时候（如gcd），可以先处理出一个数的倍数， 减去这些贡献再算这个数本身的贡献。类似dp思想，我们可以从大到小枚举，这样大数都已被处理，直接减去即可

例题总结

最大公约数

给定一个序列a[1..n]，现在要从中选择一个非空子集，显然有2^n^ - 1种选法。
求有多少个非空子集的数字的最大公约数为1。

input:

第一行包含一个正整数n(1≤n≤10^5^)。
第二行包含n个正整数a[1],a[2],…,a[n] (1≤a[i]≤10^6^)。

output:

输出一行一个整数，即答案对1e9 + 7取模的结果。

这里提供两种解法

  • 思路1
    这也是我的赛时思路。首先题目要求的是gcd不为1，也就是说不用考虑gcd到底是多少。那么我们可以筛出因子进行计算。==把每个因子看做一个限制条件==，当筛出偶数个因子的时候我们需要从答案中加上贡献，反之为奇数时要减去贡献。（可以思考选出因子1的时候。这样是0个限制条件，而有因子1的数是全体n个数，此时有2^n^ - 1种取法。ans显然是要加上这些贡献的）

    可以举个简单的例子，求100以内是3的倍数或者是5的倍数的个数，显然
    ans = 100 / 3 + 100 / 5 - 100 / 15;

    可以简单模拟一下，现在有6， 12这两个数，瞪眼法仔细思考可得答案为0。当筛因子1的时候选出两个数， 此时ans = 3；同理筛出 2，3 的时候要减去贡献，ans = ans - 6，此时ans = -3；其实你会发现6和12的gcd为6，但是他们被2和3都筛了一次。当筛因子6时，由于6有因子2， 3因此ans += qpow(2, 2) - 1;此时ans = 0与我们得出的答案一致

    这里还有一个细节，由于4 = 2^2^，被4筛其实和被2筛的性质是一样的，因此可以打上标记避免重复算

    赛时代码：

#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
#define X first
#define Y second
#define umap unordered_map
using ll = long long;
using ull = unsigned long long;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;
const int maxn = 2e6 + 5, mod = 1e9 + 7, maxp = 31, inf = 1e9;
const ll INF = 1e18;
const double eps = 1e-6;

bool isprime[maxn];
int prime[maxn];//从1开始
int ctt;

int phi[maxn];

void eulur(int x) {//这里直接拉的板子，和欧拉函数放一起了
    phi[1] = 1;
    for (int i = 2; i <= x; i++) isprime[i] = 1;
    for (int i = 2; i <= x; i++) {
        if (isprime[i]) prime[++ctt] = i, phi[i] = i - 1;
        for (int j = 1; j <= ctt && i * prime[j] <= x; j++) {
            isprime[i * prime[j]] = 0;
            if (i % prime[j] == 0) {
                phi[i * prime[j]] = phi[i] * prime[j];
                break;
            }
            else {
                phi[i * prime[j]] = phi[i] * (prime[j] - 1);
            }
        } 
    }
}

ll fastpow(ll a, ll n) {
    ll res = 1;
    a %= mod;
    while (n) {
        if (n & 1) res = res * a % mod;
        a = a * a % mod;
        n >>= 1;
    }
    return res;
}

void solve() {
    int n;
    cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    eulur(1e6 + 5);
    sort(a.begin(), a.end());
    ll ans = fastpow(2, n);//这里把1单独拿出来算了，当时想的是反着做
    ans = (ans - 1 + mod) % mod;
    const int m = a[n];//求出上限
    //下面是预处理 被因子x筛出的其实就是x的所有倍数
    //先把每个数有多少给记录在cnt数组里，再把一个数的因子个数记录在ct数组里
    vector<int> ct(m + 1), cnt(m + 1);
    vector<int> vis(m + 1);
    for (int i = 1; i <= ctt; i++) {//这里是调和奇数，时间复杂度O(nlogn)
        int p = prime[i];
        for (int j = p; j <= m; j += p) {
            ct[j]++;
            if (j % (p * p) == 0) vis[j] = 1;//给一个数的平方数打上标记
        }
    }
    
    for (int i = 1; i <= n; i++) {
        int x = a[i];
        for (int j = 1; j * j <= x; j++) {
            if (x % j) continue;
            cnt[j]++;
            if (x / j != j) cnt[x / j]++;
        }
    }
    
    for (int i = 2; i <= m; i++) {
        if (vis[i]) continue;//打上标记就跳过
        int f = 1;
        if (ct[i] % 2) f = -1;
        //f来表示对答案的贡献是加还是减
        ans += f * (fastpow(2, cnt[i]) - 1);
        ans = (ans + mod) % mod;
    }
    cout << ans << endl;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int tt = 1;
    // cin >> tt;
    while (tt--) solve();
    return 0;
}
